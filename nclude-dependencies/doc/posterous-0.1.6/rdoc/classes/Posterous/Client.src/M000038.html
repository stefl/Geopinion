    def primary_site
      res = account_info
      raise SiteError, "Couldn't find a primary site. Check login and password is valid." \
        unless res.is_a?(Hash) && res["stat"] == "ok" && res["site"]
      [res["site"]].flatten.each do |site|
        return site["id"] if site["primary"] == "true"
      end
      nil
    end
