    def build_query
      options = { :site_id    => @site_id,
                  :autopost   => @autopost,
                  :private    => @private_post,
                  :date       => @date,
                  :tags       => @tags }

      query   = { :title      => @title,
                  :body       => @body,
                  :source     => @source,
                  :sourceLink => @source_url }

      options.delete_if { |k,v| !v }
      query.merge!(options)
    end
